package main

import (
	"context"
	"fmt"
	"net/http"
	"github.com/quic-go/webtransport-go"
)

func main() {
	server := webtransport.Server{
		CheckOrigin: func(r *http.Request) bool { return true },
	}

	http.HandleFunc("/radio", func(w http.ResponseWriter, r *http.Request) {
		session, err := server.Upgrade(w, r)
		if err != nil {
			fmt.Printf("Error upgrade: %v\n", err)
			return
		}
		fmt.Println("Â¡Android conectado!")

		for {
			stream, err := session.AcceptUniStream(context.Background())
			if err != nil {
				break
			}
			go func() {
				buf := make([]byte, 4096)
				for {
					n, err := stream.Read(buf)
					if err != nil {
						break
					}
					fmt.Printf("Recibidos %d bytes\n", n)
				}
			}()
		}
	})

	fmt.Println("Servidor escuchando en puerto 8080...")
	http.ListenAndServe(":8080", nil)
}
